快速开始

Quick Start

# 安装 D&D 完整版# Install D&D full runtime $ pip install "sagasmith-dnd[documents]" $ sagasmith-dnd doctor --json # 需要语义搜索时再加 ChromaDB（~2GB）# Add ChromaDB for semantic search (~2GB) $ pip install "sagasmith-dnd[all]" $ export CHROMA_DB_PATH=~/.sagasmith/chroma_db # 启动 HTTP API（供管理前端连接）# Start HTTP API (for admin dashboard) $ pip install "sagasmith-dnd[serve]" $ sagasmith-dnd serve --port 3000
# 克隆 skill 仓库，切到 standalone 目录# Clone skill repo, switch to standalone/ $ cd standalone/ $ python portable.py doctor $ python portable.py campaign start --name "Campaign" --edition 2024 $ python portable.py roll check --dc 15 --score 16 --proficient --level 5 $ python portable.py save create --campaign "campaign_id" --label "Before boss"
# 先启动 API# Start the API first $ sagasmith-dnd serve --port 3000 # 新终端：启动 D&D 管理面板# New terminal: start D&D dashboard $ git clone https://github.com/SagaSmithAI/sagasmith-dnd-ui.git $ cd sagasmith-dnd-ui && npm install && npm run dev # 新终端：启动 CoC 守秘人面板# New terminal: start CoC Keeper dashboard $ git clone https://github.com/SagaSmithAI/sagasmith-coc-ui.git $ cd sagasmith-coc-ui && npm install && npm run dev
